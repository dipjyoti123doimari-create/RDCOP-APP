# ---------------------------------------------------------------------------
# Python packages needed for the Batching Incentive & Deduction Calculator.
#
# Install everything at once with:
#     pip install -r requirements.txt
#
# (We pin only loose minimum versions so it stays beginner-friendly and
#  easy to install. You can tighten these later if you want.)
# ---------------------------------------------------------------------------

flask                # Production web framework (replaces Streamlit)
streamlit            # Legacy UI — kept as streamlit_app.py for reference
pandas               # Reads and processes table data (Excel, calculations)
openpyxl             # Reads .xlsx Excel files and helps format Excel output
xlsxwriter           # Writes nicely formatted Excel reports (colors, freeze, etc.)
requests             # Reliable HTTP fetching (used for public Google Sheets access)
gspread              # Talks to Google Sheets (private sheets via service account)
google-auth          # Handles Google service-account login for Google Sheets
oracledb             # Connects to Oracle database (thick mode via Instant Client)
APScheduler>=3.10    # Background job scheduler (monthly auto-email)
pillow               # Generates PNG preview images embedded inline in email body
